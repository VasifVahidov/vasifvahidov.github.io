    import numpy as np
    import matplotlib.pyplot as plt
    import seaborn as sns
    import pandas as pd
    
    # List of grocery items
    items = ['Apples', 'Bananas', 'Bread', 'Milk', 'Eggs']
    
    # Prices of each item
    prices = np.array([1.20, 0.50, 2.00, 1.50, 3.00])
    
    # Frequency of purchase per week
    frequency_per_week = np.array([3, 7, 2, 5, 1])
    
    # Calculate weekly expenses
    weekly_expenses = prices * frequency_per_week
    
    # Calculate total weekly expense
    total_weekly_expense = np.sum(weekly_expenses)
    
    # Calculate monthly expenses (assuming 4 weeks in a month)
    monthly_expenses = total_weekly_expense * 4
    
    # Print out the results
    print(f"Weekly Expenses: ${total_weekly_expense:.2f}")
    print(f"Monthly Expenses: ${monthly_expenses:.2f}")
    
    # Create a DataFrame for plotting
    data = pd.DataFrame({
        'Item': items,
        'Expense': weekly_expenses
    })
    
    # Create a bar chart for weekly expenses
    plt.figure(figsize=(12, 6))
    sns.barplot(x='Item', y='Expense', data=data, palette="viridis")
    plt.title("Weekly Grocery Expenses")
    plt.xlabel("Items")
    plt.ylabel("Cost ($)")
    plt.tight_layout()
    plt.show()
    
    # Use Seaborn's style to enhance the plot
    sns.set(style="whitegrid")
    
    # Enhanced bar chart
    plt.figure(figsize=(12, 6))
    sns.barplot(x='Item', y='Expense', data=data, palette="coolwarm")
    plt.title("Enhanced Weekly Grocery Expenses")
    plt.xlabel("Items")
    plt.ylabel("Cost ($)")
    plt.tight_layout()
    plt.show()
